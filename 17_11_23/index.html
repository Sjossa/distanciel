exercise 9

let numbers = [8, 15, 22, 30, 18];

// Utiliser la méthode reduce pour calculer la somme des éléments
let somme = numbers.reduce(function (acc, current) {
    return acc + current;
}, 0); // La valeur initiale de l'accumulateur est 0

// Afficher le résultat
console.log("Somme des éléments : " + somme);
